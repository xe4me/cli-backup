Above is an example of a "Basic Info" block.

This block is loaded from a form-def.def.json file by the 'amp-block-loader' directive.
It should be the default way of loading it.

Example

In "form-def.def.json" file, in any array of blocks: 

            
    {
        "name": "basicInfo",
        "blockType": "AmpBasicInfoBlockComponent",
        "blockLayout": "INLINE",
        "commonBlock": true,
        "path": "modules/amp-basic-info-block/components/amp-basic-info-block/amp-basic-info-block.component",
        "custom": {
            ...
        }
    }

Customisation

Through the "custom" section, you can define:

  • The title of the block: 
            
    {
        "custom": {
            "blockTitle": "Tell us about yourself"
        }
    }
  • IDs of your model values: 
            
    {
        "custom": {
            "controls": [
                { "id": "myTitle" },
                { "id": "myFirstName" },
                { "id": "myMiddleName" },
                { "id": "myLastName" },
                { "id": "myDateOfBirth" }
            ]
        }
    }
  • A specific pattern for text inputs: 
            
    {
        "custom": {
            "controls": [
                {},
                {
                    "id": "FirstName",
                    "pattern": "^[A-Za-z][A-Z|a-z|'| |-]*[a-z]$FLAGS:g",
                    "errors": {
                        "pattern": "First name has an invalid value."
                    }
                }
            ]
        }
    }
  • Any error labels related to Angular invalid states: 
            
    {
        "custom": {
            "controls": [
                {},
                {},
                {
                    "id": "MiddleName",
                    "pattern": "^[A-Za-z][A-Z|a-z|'| |-]*[a-z]$FLAGS:g",
                    "errors": {
                        "required": "Middle name is a required value.",
                        "pattern": "Middle name has an invalid value."
                    }
                }
            ]
        }
    }
