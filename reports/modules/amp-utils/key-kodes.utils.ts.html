"use strict";
var KeyCodes = (function () {
    function KeyCodes() {
    }
    KeyCodes.isInputKey = function (e) {
        return (e.keyCode >= 31 && e.keyCode <= 90);
    };
    KeyCodes.isNumPadKey = function (e) {
        return (3 === e.location && e.keyCode >= 97 && e.keyCode <= 105);
    };
    KeyCodes.isNavigationKey = function (e) {
        return (KeyCodes.NAVIGATION_KEYS.indexOf(e.keyCode) !== -1);
    };
    KeyCodes.LEFT = 37;
    KeyCodes.UP = 38;
    KeyCodes.RIGHT = 39;
    KeyCodes.DOWN = 40;
    KeyCodes.BACKSPACE = 8;
    KeyCodes.SPACE = 32;
    KeyCodes.ENTER = 13;
    KeyCodes.ESCAPE = 27;
    KeyCodes.TAB = 9;
    KeyCodes.ARROWS = [37, 38, 39, 40];
    KeyCodes.LEFT_WINDOW_KEY = 91;
    KeyCodes.RIGHT_WINDOW_KEY = 92;
    KeyCodes.NAVIGATION_KEYS = [KeyCodes.SPACE, KeyCodes.ENTER, KeyCodes.UP, KeyCodes.DOWN];
    return KeyCodes;
}());
exports.KeyCodes = KeyCodes;
