"use strict";
var core_1 = require('@angular/core');
var AmpCurrencyPipe = (function () {
    function AmpCurrencyPipe() {
    }
    /*
    * @param float   inNumber:        number to be formatted
    * @param integer inDecimals:      length of decimal places
    * @param integer inIntegerGroups: number of digits to group in integer part
    */
    AmpCurrencyPipe.prototype.transform = function (inNumber, inDecimals, inIntegerGroups, inCurrencySymbol) {
        var decimals = inDecimals || 0;
        var integerGroups = inIntegerGroups || 3;
        var aNumber = (typeof (inNumber) === 'string') ? this.parseValue(inNumber) : inNumber;
        var currencySymbol = (typeof (inCurrencySymbol) === 'undefined') ? AmpCurrencyPipe.currencySymbol : inCurrencySymbol;
        var re = '\\d(?=(\\d{' + (integerGroups || 3) + '})+' + (decimals > 0 ? '\\.' : '$') + ')';
        if (isNaN(aNumber) || inNumber === null || inNumber === '') {
            return '';
        }
        else {
            // tslint:disable-next-line:no-bitwise
            return currencySymbol + aNumber.toFixed(Math.max(0, ~~decimals)).replace(new RegExp(re, 'g'), '$&,');
        }
    };
    AmpCurrencyPipe.prototype.parseValue = function (value) {
        return value ? parseFloat(value.replace(/[,\$]/g, '')) : value;
    };
    AmpCurrencyPipe.currencySymbol = '$';
    AmpCurrencyPipe = __decorate([
        core_1.Pipe({ name: 'ampCurrency' }), 
        __metadata('design:paramtypes', [])
    ], AmpCurrencyPipe);
    return AmpCurrencyPipe;
}());
exports.AmpCurrencyPipe = AmpCurrencyPipe;
