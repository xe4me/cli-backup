"use strict";
var functions_utils_1 = require('../../../modules/amp-utils/functions.utils');
var UploadStatus = (function () {
    function UploadStatus(originalName, size) {
        this.originalName = originalName;
        this.size = size;
        this.progress = {
            loaded: 0,
            total: 0,
            percent: 0,
            speed: 0,
            speedHumanized: null
        };
        this.done = false;
        this.error = false;
        this.abort = false;
        this.startTime = new Date().getTime();
        this.endTime = 0;
        this.speedAverage = 0;
        this.speedAverageHumanized = null;
    }
    UploadStatus.prototype.setProgress = function (progress) {
        this.progress = progress;
    };
    UploadStatus.prototype.setError = function () {
        this.error = true;
        this.done = true;
    };
    UploadStatus.prototype.setAbort = function () {
        this.abort = true;
        this.done = true;
    };
    UploadStatus.prototype.onFinished = function (status, statusText, response) {
        this.endTime = new Date().getTime();
        this.speedAverage = this.size / (this.endTime - this.startTime) * 1000;
        this.speedAverage = parseInt(this.speedAverage, 10);
        this.speedAverageHumanized = functions_utils_1.humanizeBytes(this.speedAverage);
        this.status = status;
        this.statusText = statusText;
        this.response = response;
        this.done = true;
    };
    return UploadStatus;
}());
exports.UploadStatus = UploadStatus;
